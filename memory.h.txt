#import <mach-o/dyld.h>
#import <mach/mach.h>

uintptr_t GetBase() { return (uintptr_t)_dyld_get_image_header(0); }

void WriteFloat(int idx, float val) {
    uintptr_t addr = GetBase() + FinalOffsets[idx];
    if (FinalOffsets[idx] != 0) { *(float*)addr = val; }
}

void Patch(int idx, const char* hex, int len) {
    uintptr_t addr = GetBase() + FinalOffsets[idx];
    if (FinalOffsets[idx] != 0) { vm_write(mach_task_self(), (vm_address_t)addr, (vm_offset_t)hex, len); }
}
